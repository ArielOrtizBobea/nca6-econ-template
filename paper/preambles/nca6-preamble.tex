}

% Confidence/Likelihood inline tags
% Usage: \confidence{very high} or \confidence{high} etc.
\newcommand{\confidence}[1]{%
  \begingroup
  \def\tmp{#1}%
  \ifx\tmp\veryhigh\colorbox{conf-veryhigh}{\textcolor{white}{\strut\,Very High Confidence\,}}%
  \else\ifx\tmp\high\colorbox{conf-high}{\textcolor{white}{\strut\,High Confidence\,}}%
  \else\ifx\tmp\medium\colorbox{conf-medium}{\strut\,Medium Confidence\,}%
  \else\ifx\tmp\low\colorbox{conf-low}{\strut\,Low Confidence\,}%
  \else\textbf{[Confidence: #1]}\fi\fi\fi\fi
  \endgroup
}
\def\veryhigh{very high}\def\high{high}\def\medium{medium}\def\low{low}

% Usage: \likelihood{very likely} etc.
\newcommand{\likelihood}[1]{\textit{\small[Likelihood: #1]}}
